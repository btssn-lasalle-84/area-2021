291  private void initialiserHandler()

293  this.handler = new Handler(this.getMainLooper())
294  {
295  @Override
296  public void handleMessage(@NonNull Message message)
297  {
298  Log.d(TAG, "[Handler] id du message = " + message.what);
299  Log.d(TAG, "[Handler] contenu du message = " + message.obj.toString());
300 
301  switch (message.what)
302  {
303  case LiaisonBluetooth.CREATION_SOCKET:
304  Log.d(TAG, "[Handler] CREATION_SOCKET = " + message.obj.toString());
305  break;
306  case LiaisonBluetooth.CONNEXION_SOCKET:
307  Log.d(TAG, "[Handler] CONNEXION_SOCKET = " + message.obj.toString());
308  afficherEtatConnexionBluetooth(message.obj.toString(),true);
309  break;
310  case LiaisonBluetooth.DECONNEXION_SOCKET:
311  Log.d(TAG, "[Handler] DECONNEXION_SOCKET = " + message.obj.toString());
312  afficherEtatConnexionBluetooth(message.obj.toString(),false);
313  break;
314  case LiaisonBluetooth.RECEPTION_TRAME:
315  if(ProtocolAREA.verifierTrameNet(message.obj.toString()))
316  {
317  toastNet.show();
318  liaisonModuleAfficheur.envoyer(ProtocolAREA.fabriquerTrameAfficheur(ProtocolAREA.TRAME_AFFICHEUR_NET, partie));
319  }
320  break;
321  }
322  }
323  };
